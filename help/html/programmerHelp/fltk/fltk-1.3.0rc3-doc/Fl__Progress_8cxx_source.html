00072     progress = (int)(w() * (value_ - minimum_) / (maximum_ - minimum_) + 0.5f);
00073   else

00076   // Draw the box and label...

00084     fl_pop_clip();
00085 
00086     labelcolor(c);
00087 
00088     if (progress<w()) {
