00054 int Fl::compose(int& del) {
00055   // character composition is now handled by the OS
00056   del = 0;
00057 #if defined(__APPLE__)
00058   // this stuff is to be treated as a function key
00059   if(Fl::e_length == 0 || Fl::e_keysym == FL_Enter || Fl::e_keysym == FL_KP_Enter || 
00060      Fl::e_keysym == FL_Tab || Fl::e_keysym == FL_Escape || Fl::e_state&(FL_META | FL_CTRL) ) {
00061     return 0;
00062   }
00063 #else
00064   unsigned char ascii = (unsigned)e_text[0];
00065   if ((e_state & (FL_ALT | FL_META)) && !(ascii & 128)) return 0;
00066 #endif
00067   if(Fl::compose_state) {
00068     del = Fl::compose_state;
00069     Fl::compose_state = 0;
00070   } else {
00071 #ifndef __APPLE__
00072     // Only insert non-control characters:
00073     if (! (ascii & ~31 && ascii!=127)) { return 0; }
00074 #endif
00075   }
00076   return 1;
