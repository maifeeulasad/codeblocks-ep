00792       i2 = j & 0xff;

00794         exit(1);

00796          = KS X 1001.1992, ignore the rest. */
00797       if (!(i1 >= 128+33 && i1 < 128+127 && i2 >= 128+33 && i2 < 128+127))
00798         continue;  /* KSC5601 specific */
00799       i1 &= 0x7f;  /* KSC5601 specific */
00800       i2 &= 0x7f;  /* KSC5601 specific */

00803       if (row < 0 || col < 0) {

00805         exit(1);
