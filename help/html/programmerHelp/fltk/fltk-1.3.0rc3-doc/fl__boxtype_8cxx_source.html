00080 void fl_frame(const char* s, int x, int y, int w, int h) {
00081   uchar *g = fl_gray_ramp();
00082   if (h > 0 && w > 0) for (;*s;) {
00083     // draw top line:
00084     fl_color(g[(int)*s++]);
00085     fl_xyline(x, y, x+w-1);
00086     y++; if (--h <= 0) break;
00087     // draw left line:
00088     fl_color(g[(int)*s++]);
00089     fl_yxline(x, y+h-1, y);
00090     x++; if (--w <= 0) break;
00091     // draw bottom line:
00092     fl_color(g[(int)*s++]);
00093     fl_xyline(x, y+h-1, x+w-1);
00094     if (--h <= 0) break;
00095     // draw right line:
00096     fl_color(g[(int)*s++]);
00097     fl_yxline(x+w-1, y+h-1, y);
00098     if (--w <= 0) break;
00099   }
