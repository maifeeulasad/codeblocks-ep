00036 static void innards(
00037     const Fl_Label* o, int X, int Y, int W, int H, Fl_Align align,
00038     int data[][3], int n)
00039 {
00040   Fl_Align a1 = align;
00041   if (a1 & FL_ALIGN_CLIP) {
00042     fl_push_clip(X, Y, W, H); a1 = (Fl_Align)(a1&~FL_ALIGN_CLIP);}
00043   fl_font((Fl_Font)o->font, o->size);
00044   for (int i = 0; i < n; i++) {
00045     fl_color((Fl_Color)(i < n-1 ? data[i][2] : o->color));
00046     fl_draw(o->value, X+data[i][0], Y+data[i][1], W, H, a1);
00047   }
00048   if (align & FL_ALIGN_CLIP) fl_pop_clip();
00049 }
