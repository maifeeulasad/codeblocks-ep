00070 static int fltk2mac(int fltk) {
00071   int a = 0;
00072   int b = sizeof(vktab)/sizeof(*vktab);
00073   while (a < b) {
00074     int c = (a+b)/2;
00075     if (vktab[c].fltk == fltk) return vktab[c].vk;
00076     if (vktab[c].fltk < fltk) a = c+1; else b = c;
00077   }
00078   return 127;
00079 }
