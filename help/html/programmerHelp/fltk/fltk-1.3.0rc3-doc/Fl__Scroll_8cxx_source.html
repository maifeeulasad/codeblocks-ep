00049   Fl_Widget** a = (Fl_Widget**)array();
00050   if (a[children()-1] != &scrollbar) {
00051     int i,j; for (i = j = 0; j < children(); j++)
00052       if (a[j] != &hscrollbar && a[j] != &scrollbar) a[i++] = a[j];
00053     a[i++] = &hscrollbar;
00054     a[i++] = &scrollbar;

00340   int dx = X-x(), dy = Y-y();
00341   int dw = W-w(), dh = H-h();

00347     Fl_Widget* o = *a++;
