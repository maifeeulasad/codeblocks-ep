00961 /* Flush the current output buffers now */
00962 void PNGAPI
00963 png_write_flush(png_structp png_ptr)
00964 {
00965    int wrote_IDAT;
00966 
00967    png_debug(1, "in png_write_flush");

00969       return;
00970    /* We have already written out all of the data */
