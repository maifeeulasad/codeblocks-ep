00037 static void rbox(int fill, int x, int y, int w, int h) {
00038   int i;
00039   int rsx ,rsy, rs;
00040   rsx = w*2/5; rsy = h*2/5;
00041   if (rsx > rsy) rs = rsy; else  rs = rsx;
00042   if (rs > RS) rs = RS;
00043   rsx = rs; rsy = rs;
00044 
00045   if (fill) fl_begin_polygon(); else fl_begin_loop();
00046   for (i=0; i<RN; i++)
00047     fl_vertex(x + offset[RN-i-1]*rsx, y + offset[i] * rsy);
00048   for (i=0; i<RN; i++)
00049     fl_vertex(x + offset[i]*rsx, y + h-1 - offset[RN-i-1] * rsy);
00050   for (i=0; i<RN; i++)
00051     fl_vertex(x + w-1 - offset[RN-i-1]*rsx, y + h-1 - offset[i] * rsy);
00052   for (i=0; i<RN; i++)
00053     fl_vertex(x + w-1 - offset[i]*rsx, y + offset[RN-i-1] * rsy);
00054   if (fill) fl_end_polygon(); else fl_end_loop();
00055 }
